# Copy Frameworks — Fórmulas de Copy de Alta Conversão
# Referência rápida para @copywriter

frameworks:

  AIDA:
    name: "AIDA"
    description: "Atenção → Interesse → Desejo → Ação"
    use_when: "Páginas de vendas, emails, ads"
    structure:
      - "A — Headline que para o scroll"
      - "I — Gancho que prende (história, dado, pergunta)"
      - "D — Benefícios + prova social + transformação"
      - "A — CTA claro e urgente"

  PAS:
    name: "PAS"
    description: "Problema → Agitação → Solução"
    use_when: "Emails, WhatsApp, copy curta"
    structure:
      - "P — Identificar a dor principal"
      - "A — Agitar a dor (consequências de não resolver)"
      - "S — Apresentar a solução (seu produto)"

  BAB:
    name: "BAB"
    description: "Before → After → Bridge"
    use_when: "Depoimentos, cases, histórias de transformação"
    structure:
      - "B — Situação ANTES (dor, frustração)"
      - "A — Situação DEPOIS (resultado, transformação)"
      - "B — A PONTE (seu método/produto)"

  4Ps:
    name: "4Ps"
    description: "Promise → Picture → Proof → Push"
    use_when: "Páginas de vendas, VSLs"
    structure:
      - "Promise — Promessa grande e específica"
      - "Picture — Pintar o cenário da transformação"
      - "Proof — Provas (depoimentos, dados, cases)"
      - "Push — Empurrão final (CTA + urgência)"

  Star_Story_Solution:
    name: "Star-Story-Solution"
    description: "Herói → História → Solução"
    use_when: "CPLs, webinars, VSLs, storytelling"
    structure:
      - "Star — Apresentar o herói (expert ou aluno)"
      - "Story — Contar a jornada (desafio → descoberta)"
      - "Solution — Revelar a solução (seu método)"

  Hook_Story_Offer:
    name: "Hook-Story-Offer"
    description: "Gancho → História → Oferta"
    use_when: "Ads, reels, stories, copy curta"
    structure:
      - "Hook — Frase que para o scroll (3 segundos)"
      - "Story — Micro-história que conecta"
      - "Offer — CTA direto com benefício"
